## USERS
curl -X POST http://localhost:8080/v1/users -H "Content-Type: application/json" -d '{"username":"test11", "password":"sadasd"}'
curl -X GET http://localhost:8080/v1/users/1
curl -X POST http://localhost:8080/v1/users/login -H "Content-Type: application/json" -d '{"username":"test11", "password":"sadasd"}'
